## Create a Self-Issued Passport (L1)

<Note>
L1/self-issued passports are intended for dev/test and low-risk reads. For production, use L2/L3 with trusted issuer policy and theft/replay protections.
</Note>

<CodeGroup>
```python Python
from uniplex import Agent

agent = Agent.create(
    agent_id="my-agent",
    permissions="*"
)

# Access the passport
passport = agent.passport
```
```typescript TypeScript
import { Agent } from 'uniplex';

const agent = await Agent.create('my-agent', {
  permissions: '*'
});

// Access the passport
const passport = agent.passport;
```
</CodeGroup>

## Verify at a Gate

<CodeGroup>
```python Python
from uniplex import Gate, TrustProfile

# L1 is dev/test only; use L2/L3 in production
gate = Gate(profile=TrustProfile.L1)

decision = gate.authorize_simple(
    passport=passport,
    action="read"
)

if decision.allowed:
    # Proceed with operation
    pass
else:
    print(f"Denied: {decision.reason_code}")
```
```typescript TypeScript
import { Gate, TrustProfile } from 'uniplex';

// L1 is dev/test only; use L2/L3 in production
const gate = new Gate({ profile: TrustProfile.L1 });

const decision = await gate.authorizeSimple(passport, 'read');

if (decision.allowed) {
  // Proceed with operation
} else {
  console.log(`Denied: ${decision.reasonCode}`);
}
```
</CodeGroup>

## Quick Check with Agent

The simplest way to check authorization:

<CodeGroup>
```python Python
from uniplex import Agent

agent = Agent.create("my-agent", permissions="mcp:*")

# Quick check (no attestation)
if agent.can("mcp:search"):
    print("Allowed!")

# Full authorization (with decision object)
decision = agent.authorize("mcp:search")
```
```typescript TypeScript
import { Agent } from 'uniplex';

const agent = await Agent.create('my-agent', { permissions: 'mcp:*' });

// Quick check (no attestation)
if (agent.can('mcp:search')) {
  console.log('Allowed!');
}

// Full authorization (with decision object)
const decision = await agent.authorize('mcp:search');
```
</CodeGroup>

## MCP Integration

<CodeGroup>
```python Python
from uniplex import MCPAuthorizer, TrustProfile

authorizer = MCPAuthorizer(
    server_id="mcp://mytools",
    profile=TrustProfile.L1
)

def handle_tool_call(passport, tool_name, params):
    result = authorizer.authorize(passport, tool_name, parameters=params)
    
    if not result.allowed:
        raise PermissionError(f"Denied: {result.reason_code}")
    
    # Proceed with tool execution
    return {"status": "ok"}
```
```typescript TypeScript
import { MCPAuthorizer, TrustProfile } from 'uniplex';

const authorizer = new MCPAuthorizer({
  serverId: 'mcp://mytools',
  profile: TrustProfile.L1
});

async function handleToolCall(passport, toolName, params) {
  const result = await authorizer.authorize(passport, toolName, {
    parameters: params
  });
  
  if (!result.allowed) {
    throw new Error(`Denied: ${result.errorCode}`);
  }
  
  // Proceed with tool execution
  return { status: 'ok' };
}
```
</CodeGroup>
